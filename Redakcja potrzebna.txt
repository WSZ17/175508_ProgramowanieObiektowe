main(String[] args) {        Scanner scanner = new Scanner(System.in);        System.out.print("Podaj pojedynczy znak: ");        char znak = scanner.next().charAt(0);        System.out.print("Podaj wysokość piramidy (n): ");        int n = scanner.nextInt();        StringBuilder piramidka = new StringBuilder();        for (int i = 1; i <= n; i++) {            for (int j = 0; j < n - i; j++) {                piramidka.append(" ");            }            for (int k = 0; k < 2 * i - 1; k++) {                piramidka.append(znak);            }            piramidka.append("\n");        }        System.out.println(piramidka.toString());    }}
